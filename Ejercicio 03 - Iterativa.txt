#include<iostream>
#include<conio.h>

using namespace std;

void orden(int tam,int num[]){
	int p;
	
	for (int i = 0;i < tam; i++){
		for (int j = 0; j< tam-1; j++){
			if (num[j] < num[j+1]){ 
				p = num[j]; 
				num[j] = num[j+1]; 
				num[j+1] = p;
			}
		}
	}
}


int main(){
	int num[5] = {1,2,3,4,5}; 
	int tam=5;
	cout<<"orden ascendente: "<<endl;
	orden(tam,num);	
	for (int i = tam-1; i >= 0; i--) 
		cout << num[i] << endl;
	return 0;
}
