#include<iostream>
#include<conio.h>

using namespace std;

void ascendente( int num[], int i, int p, int men, int tam){
	if( i <= tam-1 ){
	   for( int j = i ; j <= tam-1; j++ ){
	   	men = num[ i ];
	   	
			if( men > num[ j ] ){
	   		men = num[ j ];
	   		p = j;
	   		num[ p ] = num[ i ];
		      num[ i ] = men;
			}
		}
	   ascendente( num, i + 1, p, men, tam);
	}	
}



int main(){
	int num[] = { 1,2,3,4,5};
	int tam=5;
	int men = num[ 0 ];
	int p;
	ascendente( num, 0 ,p , men,tam);
	
	for( int i = 0; i <= tam-1; i++ ){
		cout << num[ i ] << endl;
	}
	
	return 0;
}
